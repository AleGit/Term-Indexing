\begin{exampleblock}{Positions of a term}
$
	\Pos(t) =  
	\begin{cases}
		\{ \epsilon \} & \text{if }t=x\in\SIGV \\
		\{ \epsilon \} \cup \{ ip\mid 1\leq i\leq n \land p\in\Pos(t_i)\} &
		\text{if }t=f(t_1,\ldots,t_n)
	\end{cases}
$
\end{exampleblock}

\begin{example}[forward subsumption]
			\vspace{-1em}
	\begin{gather*}
		S = \{ \TI{1}\mP(x,y), \TI{2}\lnot \mP(\ma,z)\} \cup \{\TI{3}\colG\mP(\ma,z') \}
		\tag*{$t_1$ subsumes $t_3$}
		\\[0.7em]
		\infer[\{x\mapsto\ma,y\mapsto z\}]
		{\square}
		{\mP(x,y) & \lnot \mP(\ma,z)}
		\tag*{Resolution}
		\\[0.7em]
		S\bot = \{ \mP(\bot,\bot), {\colLo\lnot \mP(\ma,\bot), \mP(\ma,\bot)} \}
		\tag*{InstGen / SMT}
	\end{gather*}
	
\end{example}

\section{Saturation-based theorem provers}

\frame{
	\frametitle{Saturation-based theorem provers}
	\framesubtitle{General precedure}	
	\begin{enumerate}
	\item Start with a set of clauses
	\item Generate new clauses
	\item Add generated clauses to the set
	\item Stop if 
	\begin{enumerate}
	\item set is saturated, i.e.~no new clauses were generated
	\item proof for unsatisfiability has been generated
	\item time is up (or space is out)
	\end{enumerate}
	Otherwise, continue with 2.
	\end{enumerate}
}

\subsection{Optimized}
\frame{
	\frametitle{Saturation-based theorem provers}	
	\framesubtitle{Improved procedure}
	\begin{enumerate}
	\item Start with a set of clauses
	\item Generate new clauses with suitable \MYEM{inference} rules
	\item Add generated \MYEM{non-redundant} clauses to the set
	\item Stop if 
	\begin{enumerate}
	\item set is saturated, i.e.~no non-redundant clauses were generated
	\item proof for unsatisfiability has been found
	\item time is up (or space is out)
	\end{enumerate}
	Otherwise, continue with 2.
	\end{enumerate}
}

\begin{itemize}
\item Find terms that are variants of a given term.
$\vari(\ell,t) \Leftrightarrow \exists\sigma\ \ell\sigma = t$ and $\sigma$ is renaming.
\item Find terms that are unifiable with a given term. 
$\unif(\ell,t) \Leftrightarrow \exists\sigma\ \ell\sigma = t\sigma$
\item Find terms that are instances of a given term.
$\inst(\ell,t) \Leftrightarrow \exists\sigma\ \ell = t\sigma$
\item Find terms that are generalizations of a given term.
$\gene(\ell,t) \Leftrightarrow \exists\sigma\ \ell\sigma = t$
\end{itemize}
